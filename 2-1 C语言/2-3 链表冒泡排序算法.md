void soert_list(PNODE pHead)
{
	int i,j,length;
	int tmp;
	PNODE ptr = pHead->pNext;
	length = length_list(pHead);
	for(i = 0;i < length;i++)
	{
		for(j = i;j < length-1;j++)//少比较一个
		{
			if(ptr->data < ptr->pNext->data)
			{
				tmp = ptr->pNext->data;
				ptr->pNext->data = ptr->data;
				ptr->data = tmp; 
			}		
			ptr = ptr->pNext;	
		}
		ptr = pHead->pNext;//指针回头
	}
}
