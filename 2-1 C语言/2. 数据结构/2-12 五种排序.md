void bubble_sort(int * arr,int len)
{
    int i,j,t;
    for(i = 0; i < len; i++)
    {
        for(j = i+1; j < len-1; j++)
        {
            if(arr[i] > arr[j])
            {
                t = arr[i];
                arr[i] = arr[j];
                arr[j] = t;
            }
        }
    }
}
