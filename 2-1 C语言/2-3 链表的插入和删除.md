## 插入

```c
typedef struct Node
{
	int data;
	struct Node *pNext;
}NODE,*PNODE;

bool insert_list(PNODE pHead,int pos,int val)
{
    int i = 0;
    PNODE p = pHead;
    
    while(NULL != p&&i<pos-1)
    {
        p = p->pNext;
        ++i;
    }
    
    if(i>pos-1 || NULL==p)
		return false;
    
    PNODE pNew = (PNODE)malloc(sizeof(NODE));
    if(NULL == pNew)
    {
        printf("动态内存分配失败！\n")；
        exit(-1);
    }
    pNew->data = val;
    PNODE q = p->pNext;
    p-pNext = pNew;
    pNew->pNext = q;
    
    return true;
}
```

## 删除

```c
typedef struct Node
{
	int data;
	struct Node *pNext;
}NODE,*PNODE;

bool delete_list(PNODE,int pos,int* pVal)
{
    int i = 0;
    PNODE p = pHead;
    
    while(NULL != p->pNext&&i<pos-1)
    {
        p = p->pNext;
        ++i;
    }
    
    if(i>pos-1 || NULL==p->pNext)
		return false;
    
	PNODE q = p->pNext;
    *pVal = q->data;
    
    //删除p节点后面的节点
    p->pNext = p->pNext-pNext;
    free(q);
    q = NULL;
    
    return true;    
}
```
